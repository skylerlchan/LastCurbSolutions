Designed for Developers

Simple, secure, and machine-readable API to access curb availability data. Integrate directly into autonomous systems or delivery routing software.

const coords = { lat: 37.7749, lon: -122.4194 };

const res = await fetch(`https://api.lastcurb.com/v1/curbs?lat=${coords.lat}&lon=${coords.lon}`, {
  headers: {
    "Authorization": "Bearer YOUR_API_KEY"
  }
});
const data = await res.json();

if (data.available) {
  console.log("Curb available at:", data.spot);
} else {
  console.log("No curb available nearby. Rerouting...");
}

import requests

coords = {"lat": 37.7749, "lon": -122.4194}
headers = {"Authorization": "Bearer YOUR_API_KEY"}

res = requests.get("https://api.lastcurb.com/v1/curbs", params=coords, headers=headers)
data = res.json()

if data["available"]:
    print("Curb available at:", data["spot"])
else:
    print("No curb available nearby. Rerouting...")
